A. YES or YES?

把输入字符串全部大写后与字符串YES比较即可。

#include <iostream>
#include <string>

#define OPTIMIZE() do {             \
    ios::sync_with_stdio(false);    \
    cin.tie(nullptr);               \
} while (0);

using namespace std;

string lower(const string& s) 
{
    string t = s;
    int len = t.size();

    for (int i = 0; i < len; ++i) {
        t[i] = tolower(t[i]);
    }

    return t;
}

int main()
{
    int tt;
    cin >> tt;

    while (tt-- > 0) {
        string s;

        cin >> s;
        cout << (lower(s) == "yes" ? "YES" : "NO") << endl;
    }
    
    return 0;
}

B. ICPC Balloons

对于所有题目，维护一个数组来存储通过人数，并且开一个变量ans存储发出去的气球数。在更新数组时，如果是第一次通过，则把ans加 222，否则加 111。

#include <iostream>
#include <map>
#include <string>

#define OPTIMIZE() do {             \
    ios::sync_with_stdio(false);    \
    cin.tie(nullptr);               \
} while (0);

using namespace std;

int solve(int len, const string& s)
{
    int balloons = 0;
    map<char, int> problems;

    for (const auto& problem : s) {
        if (problems[problem] == 0) {
            balloons++;
        } 

        problems[problem]++;
        balloons++;
    }

    return balloons;
}

int main()
{
    int tt;
    cin >> tt;

    while (tt-- > 0) {
        int len;
        string s;

        cin >> len >> s;
        cout << solve(len, s) << endl;
    }
    
    return 0;
}

C. Cypher

在每一位上根据拨动顺序反推即可。

#include <iostream>
#include <string>
#include <vector>

#define OPTIMIZE() do {             \
    ios::sync_with_stdio(false);    \
    cin.tie(nullptr);               \
} while (0);

using namespace std;

inline int up(int wheel) 
{
    return (wheel + 1) % 10;
}

inline int down(int wheel) 
{
    return (wheel == 0 ? 9 : wheel - 1);
}

vector<int> solve(int wheels_cnt, vector<int> wheels, const vector<string>& ops)
{
    for (int i = 0; i < wheels_cnt; i++) {
        for (const auto& op : ops[i]) {
            if (op == 'D') {
                wheels[i] = up(wheels[i]);
            } else {
                wheels[i] = down(wheels[i]);
            }
        }
    }

    return wheels;
}

int main()
{
    int tt;
    cin >> tt;

    while (tt-- > 0) {
        int wheels_cnt;
        vector<string> ops;
        vector<int> wheels, ans;

        cin >> wheels_cnt;

        ops.resize(wheels_cnt);
        wheels.resize(wheels_cnt);

        for (auto& wheel : wheels) {
            cin >> wheel;
        }

        for (auto& op : ops) {
            int len;
            cin >> len >> op;
        }

        ans = solve(wheels_cnt, wheels, ops);

        for (const auto& val : ans) {
            cout << val << ' ';
        }

        cout << endl;
    }
    
    return 0;
}
