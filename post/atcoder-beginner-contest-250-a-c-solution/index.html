AtCoder Beginner Contest 250 A-C 题解

2022-05-21 AtCoder

A. Adjacent Squares

此题只需暴力枚举所有长方形，然后挨个判断这个长方形是否与 (R,C)(R, C)(R,C) 相邻即可。

// AtCoder Beginner Contest 250 - A. Adjacent Squares
#include <iostream>

#define OPTIMIZE() do {             \
    ios::sync_with_stdio(false);    \
    cin.tie(nullptr);               \
    cout.tie(nullptr);              \
} while (0);

using namespace std;

// 判断是否相邻
inline bool check(int a, int b, int c, int d) {
    return abs(a - c) + abs(b - d) == 1;
}

int main() {
    OPTIMIZE();

    int h, w, r, c, ans = 0;
    cin >> h >> w >> r >> c;
    
    // 枚举所有长方形
    for (int i = 1; i <= h; i++) {
        for (int j = 1; j <= w; j++) {
            if (check(i, j, r, c)) {
                ans++;
            }
        }
    }
    
    cout << ans << endl;
    return 0;
}

B. Enlarged Checker Board

如果我们把瓷砖看作一个 1×11 \times 11×1 的涂色的方格，其中方格上涂的色代表对应的瓷砖上涂的色，则我们可以用一个 N×NN \times NN×N 的二维数组来存储所有瓷砖的颜色和位置。随后，我们需要根据这个二维数组得出最终输出的结果。

// AtCoder Beginner Contest 250 - B. Enlarged Checker Board
#include <iostream>

#define OPTIMIZE() do {             \
    ios::sync_with_stdio(false);    \
    cin.tie(nullptr);               \
    cout.tie(nullptr);              \
} while (0);

using namespace std;

int n, a, b;
char tiles[15][15], board[105][105];

inline char reverse_color(char color) {
    return color == '.' ? '#' : '.';
}

void paint(int n) {
    for (int i = 0; i < n; i++) {
        if (i % 2 == 0) {
            tiles[i][0] = '.';
        } else {
            tiles[i][0] = '#';
        }

        for (int j = 1; j < n; j++) {
            tiles[i][j] = reverse_color(tiles[i][j - 1]);
        }
    }
}

int main() {
    OPTIMIZE();

    cin >> n >> a >> b;
    paint(n);

    for (int i = 0; i < a * n; i += a) {
        for (int j = 0; j < b * n; j += b) {
            for (int k = 0; k <= a; k++) {
                for (int l = 0; l <= b; l++) {
                    board[i + k][j + l] = tiles[i / a][j / b];
                }
            }
        }
    }

    for (int i = 0; i < a * n; i++) {
        for (int j = 0; j < b * n; j++) {
            cout << board[i][j];
        }

        cout << endl;
    }
    
    return 0;
}

C. Adjacent Swaps

因为我们需要交换写着某个数字的球，而不是某个位置的球，所以我们使用map<int, int>存储所有球所在的位置。在进行交换操作时，不仅要把这两个球交换，还要相应地更新它们的位置。

// AtCoder Beginner Contest 250 - C. Adjacent Swaps
#include <iostream>
#include <map>

#define OPTIMIZE() do {             \
    ios::sync_with_stdio(false);    \
    cin.tie(nullptr);               \
    cout.tie(nullptr);              \
} while (0);

using namespace std;

int main() {
    OPTIMIZE();

    int n, Q;
    int balls[200005];
    map<int, int> pos;    // 记录每个球所在的位置

    cin >> n >> Q;

    for (int i = 1; i <= n; i++) {
        balls[i] = i;
        pos[balls[i]] = i;
    }

    while (Q--) {
        int x;
        cin >> x;

        int a = pos[x], b;    // 需要交换位置的两个球的位置

        if (a == n) {
            b = a - 1;
        } else {
            b = a + 1;
        }

        swap(balls[a], balls[b]);
        swap(pos[balls[a]], pos[balls[b]]);
    }

    for (int i = 1; i <= n; i++) {
        cout << balls[i] << " ";
    }
    
    return 0;
}
