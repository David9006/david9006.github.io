A. Growth Record

此题就是求公差为 DDD 、长度为 XXX 的等差数列的第 MMM 项。如果 M>XM > XM>X，则答案是最后一项。

N, M, X, T, D = map(int, input().split())

for i in range(N, M - 1, -1):
    if i == M:
        print(T)
        break
        
    if i <= X:
        T -= D

B. Counterclockwise Rotation

可以让点 (a,b)(a, b)(a,b) 在极坐标中旋转 ddd 度。只需把点 (a,b)(a, b)(a,b) 的坐标转换成极坐标，旋转 ddd 度，再把旋转后的点的坐标转换回来，就可以了。

#include <cmath>
#include <iomanip>
#include <iostream>
#include <vector>

#define OPTIMIZE() do {             \
    ios::sync_with_stdio(false);    \
    cin.tie(nullptr);               \
} while (0);

using namespace std;

const double PI = 3.14159265358979323846;

void polar_to_rect(double r, double theta, double &x, double &y) {
    x = r * cos(theta);
    y = r * sin(theta);
}

void rect_to_polar(double x, double y, double &r, double &theta) {
    r = sqrt(x * x + y * y);
    theta = atan2(y, x);
}

int main() {
    int d;
    double a, b, r, theta;

    cin >> a >> b >> d;

    rect_to_polar(a, b, r, theta);
    theta += d * PI / 180;
    polar_to_rect(r, theta, a, b);

    cout << fixed << setprecision(6) << a << " " << b << endl;
    return 0;
}

C. XX to XXX

考虑把字符串 SSS 表示成很多重复字符组成的“块”的组合。

#include <iostream>
#include <string>
#include <vector>

#define OPTIMIZE() do {             \
    ios::sync_with_stdio(false);    \
    cin.tie(nullptr);               \
} while (0);

using namespace std;

typedef pair<char, int> chunk_t;

vector<chunk_t> generate_chunks(string s) {
    vector<chunk_t> chunks;

    for (const auto& c : s) {
        if (!chunks.empty() && c == chunks.back().first) {
            chunks.back().second++;
        } else {
            chunks.emplace_back(c, 1);
        }
    }

    return chunks;
}

int main()
{
    OPTIMIZE();

    string s, t;
    vector<chunk_t> chunks_s, chunks_t;

    cin >> s >> t;

    chunks_s = generate_chunks(s);
    chunks_t = generate_chunks(t);

    if (chunks_s.size() != chunks_t.size()) {
        cout << "No" << endl;
        return 0;
    }

    int len = chunks_s.size();

    for (int i = 0; i < len; i++) {
        if (chunks_s[i].first != chunks_t[i].first || chunks_s[i].second > chunks_t[i].second) {
            cout << "No" << endl;
            return 0;
        }

        if (chunks_s[i].second < 2 && chunks_t[i].second >= 2) {
            cout << "No" << endl;
            return 0;
        }
    }
    
    cout << "Yes" << endl;
    return 0;
}
